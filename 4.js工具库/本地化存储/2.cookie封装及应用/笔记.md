# cookie 的封装及拖拽应用

```js
const manageCookie = {
  //设置cookie
  set(name, value, date) {
    document.cookie = `${name}=${value}; max-age=${date}`;
  },
  //移除cookie
  remove(name) {
    this.set(name, "", 0);
  },
  //获取cookie
  get(name) {
    document.cookie.split("; ").forEach((index) => {
      let item = index.split("=");
      if (item[0] === name) {
        return item[1];
      }
    });
  },
};
```
